10  public class BasicShapeRenderer {
11  
12      /**
13       * 
14       * @param cx center x.
15       * @param cy center y.
16       * @param r radius.
17       * @param segments segments.
18       * @param red red.
19       * @param g green.
20       * @param b blue.
21       */
22      public static void drawFilledCircle(float cx, float cy, float r, int segments, float red, float g, float b) {
23          double theta = 2 * 3.1415926 / (double) segments;
24          double c = Math.cos(theta);
25          double s = Math.sin(theta);
26          double t;
27  
28          float x = r;
29          float y = 0;
30  
31          GL11.glColor3f(red, g, b);
32  
33          GL11.glBegin(GL11.GL_TRIANGLE_FAN);
34          for (int ii = 0; ii < segments; ii++) {
35              GL11.glVertex2f(x + cx, y + cy);//output vertex
36  
37              //apply the rotation matrix
38              t = x;
39              x = (float) (c * x - s * y);
40              y = (float) (s * t + c * y);
41          }
42          GL11.glEnd();
43      }

45  }
