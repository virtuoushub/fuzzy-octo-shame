17       * If floating point numbers are close enough in value.
18       *
19       * @param a floating point number
20       * @param b floating point number
21       * @return if a is close enough to b
22       * @see <a href="http://stackoverflow.com/a/6837237">comparing float/double values using == operator</a>
23       */
24      public static boolean compareFloats(final float a, final float b) {
25          return Math.abs(a / b - 1) < MACHINE_EPSILON;
26      }
