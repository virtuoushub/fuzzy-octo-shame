3   import org.lwjgl.opengl.GL11;
4   import org.newdawn.slick.SlickException;
5   import org.newdawn.slick.opengl.renderer.Renderer;
6   
7   import javax.annotation.Nullable;

9   /**
10   * A utility to allow performing GL operations without contaminating the
11   * Slick OpenGL state. Note this will not protect you from OpenGL programming errors
12   * like a glBegin() without a glEnd(), or glPush() without glPop() etc.
13   * 
14   * Expected usage:
15   * 
16   * <code>
17   * SlickCallable callable = new SlickCallable() {
18   *        public performGLOperations() throws SlickException {
19   *             GL.glTranslate(0,0,1);
20   *             glBegin(GL.GL_POLYGONS);
21   *              glVertex(..);
22   *              ...
23   *          glEnd();
24   *     }
25   * }
26   * callable.call();
27   * </code>
28   * 
29   * Alternatively you can use the static methods directly
30   * 
31   * <code>
32   * SlickCallable.enterSafeBlock();
33   * 
34   * GL.glTranslate(0,0,1);
35   * glBegin(GL.GL_POLYGONS);
36   *     glVertex(..);
37   *     ...
38   * glEnd();
39   * 
40   * SlickCallable.leaveSafeBlock();
41   * </code>
42   * 
43   * @author kevin
44   */
45  public abstract class SlickCallable {
46      /** The last texture used */
47      @Nullable
48      private static Texture lastUsed;
49      /** True if we're in a safe block */
50      private static boolean inSafe = false;
51  
52  
53      /**
54       * Enter a safe block ensuring that all the OpenGL state that slick
55       * uses is safe before touching the GL state directly.
56       */
57      public static void enterSafeBlock()
58      {
59          if (inSafe) {
60              return;
61          }
62  
63          Renderer.get().flush();
64          lastUsed = TextureImpl.getLastBind();
65          TextureImpl.bindNone();
66          GL11.glPushAttrib(GL11.GL_ALL_ATTRIB_BITS);
67          GL11.glPushClientAttrib(GL11.GL_ALL_CLIENT_ATTRIB_BITS);
68          GL11.glMatrixMode(GL11.GL_MODELVIEW);
69          GL11.glPushMatrix();
70          GL11.glMatrixMode(GL11.GL_PROJECTION);
71          GL11.glPushMatrix();
72          GL11.glMatrixMode(GL11.GL_MODELVIEW);
73  
74          inSafe = true;
75      }
76  
77      /**
78       * Leave a safe block ensuring that all of Slick's OpenGL state is
79       * restored since the last enter.
80       */
81      public static void leaveSafeBlock()
82      {
83          if (!inSafe) {
84              return;
85          }
86  
87          GL11.glMatrixMode(GL11.GL_PROJECTION);
88          GL11.glPopMatrix();
89          GL11.glMatrixMode(GL11.GL_MODELVIEW);
90          GL11.glPopMatrix();
91          GL11.glPopClientAttrib();
92          GL11.glPopAttrib();
93  
94          if (lastUsed != null) {
95              lastUsed.bind();
96          } else {
97              TextureImpl.bindNone();
98          }
99  
100         inSafe = false;
101     }
102 
103     /**
104      * Cause this callable to perform it's GL operations (@see performGLOperations()). This
105      * method will block until the GL operations have been performed.
106      *
107      * @throws SlickException Indicates a failure while performing the GL operations or
108      * maintaing SlickState
109      */
110     public final void call() {
111         enterSafeBlock();
112 
113         performGLOperations();
114 
115         leaveSafeBlock();
116     }
117 
118     /**
119      * Perform the GL operations that this callable is intended to. This operations should
120      * not effect the slick OpenGL state.
121      *
122      * @throws SlickException Indicates a failure of some sort. This is user exception
123      */
124     protected abstract void performGLOperations();
125 }
