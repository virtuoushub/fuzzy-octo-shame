3   /**
4    * Utility to handle Java's odd trig performance issues
5    *
6    * @author JeffK
7    */
8   public class FastTrig {
9      /**
10       * Fast Trig functions for x86.
11       * This forces the trig functiosn to stay within the safe area on the x86 processor (-45 degrees to +45 degrees)
12       * The results may be very slightly off from what the Math and StrictMath trig functions give due to
13       * rounding in the angle reduction but it will be very very close.
14       *
15       * @param radians The original angle
16       * @return The reduced Sin angle
17       */
18      private static double reduceSinAngle(double radians) {
19          radians %= Math.PI * 2.0; // put us in -2PI to +2PI space
20          if (Math.abs(radians) > Math.PI) { // put us in -PI to +PI space
21              radians = radians - (Math.PI * 2.0);
22          }
23          if (Math.abs(radians) > Math.PI / 2) {// put us in -PI/2 to +PI/2 space
24              radians = Math.PI - radians;
25          }
26  
27          return radians;
28      }
29  
30      /**
31       * Get the sine of an angle
32       *
33       * @param radians The angle
34       * @return The sine of the angle
35       */
36      public static double sin(double radians) {
37          radians = reduceSinAngle(radians); // limits angle to between -PI/2 and +PI/2
38          if (Math.abs(radians) <= Math.PI / 4) {
39              return Math.sin(radians);
40          } else {
41              return Math.cos(Math.PI / 2 - radians);
42          }
43      }
44  
45      /**
46       * Get the cosine of an angle
47       *
48       * @param radians The angle
49       * @return The cosine of the angle
50       */
51      public static double cos(double radians) {
52          return sin(radians + Math.PI / 2);
53      }
54  
55  }
