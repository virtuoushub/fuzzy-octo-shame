1   package org.newdawn.slick.util;
2   
3   import javax.annotation.Nonnull;
4   import javax.annotation.Nullable;
5   import java.io.InputStream;
6   import java.net.URL;
7   
8   /**
9    * A resource location that searches the classpath
10   * 
11   * @author kevin
12   */
13  public class ClasspathLocation implements ResourceLocation {
14      /**
15       * @see org.newdawn.slick.util.ResourceLocation#getResource(java.lang.String)
16       */
17      @Nullable
18      public URL getResource(@Nonnull String ref) {
19          String cpRef = ref.replace('\\', '/');
20          return ResourceLoader.class.getClassLoader().getResource(cpRef);
21      }
22  
23      /**
24       * @see org.newdawn.slick.util.ResourceLocation#getResourceAsStream(java.lang.String)
25       */
26      public InputStream getResourceAsStream(@Nonnull String ref) {
27          String cpRef = ref.replace('\\', '/');
28          return ResourceLoader.class.getClassLoader().getResourceAsStream(cpRef);
29      }
30  
31  }
