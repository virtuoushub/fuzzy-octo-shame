1   package org.newdawn.slick.loading;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import org.newdawn.slick.opengl.InternalTextureLoader;
7   import org.newdawn.slick.util.Log;
8   
9   import javax.annotation.Nonnull;
10  import javax.annotation.Nullable;
11  
12  /**
13   * A central list where all deferred loading resoures end up when deferred loading is in use. Each
14   * texture and sound loaded will be put in this list and can be loaded in one by one 
15   *
16   * @author kevin
17   */
18  public class LoadingList {
19      /** The single instance of this list */
20      @Nonnull
21      private static LoadingList single = new LoadingList();
22  
23      /**
24       * Get the single global loading list
25       *
26       * @return The single global loading list
27       */
28      @Nonnull
29      public static LoadingList get() {
30          return single;
31      }
32  
33      /**
34       * Indicate if we're going to use deferred loading. (Also clears the current list)
35       *
36       * @param loading True if we should use deferred loading
37       */
38      public static void setDeferredLoading(boolean loading) {
39          single = new LoadingList();
40  
41          InternalTextureLoader.get().setDeferredLoading(loading);
42          /** pc 2014-11-10 START commenting out as I am not currently using org.newdawn.slick.openal.SoundStore
43           SoundStore.get().setDeferredLoading(loading);
44           pc 2014-11-10 END
45           */
46      }
47  
48      /**
49       * Check if we're using deferred loading
50       *
51       * @return True if we're using deferred loading
52       */
53      public static boolean isDeferredLoading() {
54          return InternalTextureLoader.get().isDeferredLoading();
55      }
56  
57      /** The list of deferred resources to load */
58      @Nonnull
59      private final List<DeferredResource> deferred = new ArrayList<>();
60      /** The total number of elements that have been added - does not go down as elements are removed */
61      private int total;
62  
63      /**
64       * Create a new list
65       */
66      private LoadingList() {
67      }
68  
69      /**
70       * Add a resource to be loaded at some later date
71       *
72       * @param resource The resource to be added
73       */
74      public void add(DeferredResource resource) {
75          total++;
76          deferred.add(resource);
77      }
78  
79      /**
80       * Remove a resource from the list that has been loaded for
81       * other reasons.
82       *
83       * @param resource The resource to remove
84       */
85      public void remove(@Nonnull DeferredResource resource) {
86          Log.info("Early loading of deferred resource due to req: "+resource.getDescription());
87          total--;
88          deferred.remove(resource);
89      }
90  
91      /**
92       * Get the total number of resources that were in the list originally
93       *
94       * @return The total number of resources that were in the list originally
95       */
96      public int getTotalResources() {
97          return total;
98      }
99  
100     /**
101      * Get the number of remaining resources
102      *
103      * @return The number of resources that still need to be loaded
104      */
105     public int getRemainingResources() {
106         return deferred.size();
107     }
108 
109     /**
110      * Get the next resource that requries loading
111      *
112      * @return The next resource to load or null if there are no more remaining
113      */
114     @Nullable
115     public DeferredResource getNext() {
116         if (deferred.size() == 0) {
117             return null;
118         }
119 
120         return deferred.remove(0);
121     }
122 }
